使用socket实现简单的在线聊天室

效果图

server

import socket
import threading

server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind(("0.0.0.0",8000))
server.listen(5)

user_list = []
user_dict = {}

def tell_all_user(sock, msg):
    current_sock = sock.fileno()
    for u in user_list:
        if u.fileno() != current_sock: # 除了当前的sock，向其他的sock链接分发消息
            try:
                u.send(msg.encode("utf8"))
            except :
                pass
        else:
            pass

def create_user_thread(sock):
    current_sock = sock.fileno()
    nickname = sock.recv(1024).decode("utf8")
    user_dict[current_sock] = nickname
    user_list.append(sock)
    print('connection', current_sock, ' has nickname :', nickname)
    tell_all_user(sock, "【系统提示】用户"+nickname+"进入聊天室")
    while True:
        try:
            msg = sock.recv(1024).decode("utf8")
            if msg:
                print(nickname + "：" + msg)
                tell_all_user(sock, nickname + "：" + msg)
        except :
            user_list.remove(sock)
            tell_all_user(sock, "【系统提示】用户"+nickname+"退出聊天室")
            print(nickname, 'exit, ', len(user_list), ' person left')
            sock.close()
            break


while True:
    sock, addr = server.accept()
    print('Accept a new connection', sock.getsockname(), sock.fileno())
    try:
        sock.send("welcome to server!".encode('utf8'))
        thread = threading.Thread(target=create_user_thread, args=(sock,))
        thread.start()
    except:
        pass

import socket
import threading

client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
client.connect(("127.0.0.1", 8000))
banner = client.recv(1024).decode("utf8")
print(banner)
nickname = input("请输入你的昵称")
client.send(nickname.encode("utf8"))

def recv_func():
    while True:
        try:
            data = client.recv(1024).decode("utf8")
            if data:
                print(data)
            else:
                pass
        except :
            print("server closed")
            break

def send_dunc():
    while True:
        try:
            msg = input().encode("utf8")
            client.send(msg)
        except :
            print("server closed")
            break

send_thread = threading.Thread(target=send_dunc, daemon=True)
recv_thread = threading.Thread(target=recv_func, daemon=True)

thread_list = [send_thread, recv_thread]
for thread in thread_list:
    thread.start()
thread.join()

socket发送http请求

# 引入urllib 只是为了解析url方便，也可以手写
from urllib.parse import urlparse
import socket

def request(url):
    url = urlparse(url)
    host = url.netloc
    path = url.path
    if path == "":
        path = "/"

    client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    client.connect((host, 80))
    reques_header = "GET {} HTTP/1.1\r\nHost: {}\r\nConnection: close\r\n\r\n".format(path, host)
    client.send(reques_header.encode())

    d = b""

    while True:
        data = client.recv(1024)
        if data:
            d += data
        else:
            break
    print(d.decode())

if __name__ == "__main__":
    request("http://www.baidu.com")

简单的netcat(python3）

import argparse
import socket
import threading
import subprocess
import sys


def init():
    args = argparse.ArgumentParser()
    args.add_argument("-t", "--target", help="target host")
    args.add_argument("-p", "--port", required=True, help="target port", type=int)
    args.add_argument("-l", "--listen", action="store_true", help="listen on [host]:[port] for incoming connections")
    args.add_argument("-c", "--command", action="store_true", help="initialize a command shell")
    args.add_argument("-u", "--upload", help="upon receiving connection upload a file and write to [destination]")
    args.add_argument("-e", "--execute", help="execute the given file upon receiving a connection")
    options = args.parse_args()
    return options


def run_command(command):
    try:
        #创建一个子进程执行系统命令
        output = subprocess.check_output(command, stderr=subprocess.STDOUT, shell=True)
    except:
        output = "Failed to execute command.".encode("utf8")
    return output


def client_handler(client_socket):
    #文件上传
    if options.upload:
        file_butter = ""
        while True:
            data = client_socket.recv(1024).decode("utf8")
            #遇到输入#EOF#就结束
            if "#EOF#" in data:
                client_socket.send("end upload".encode("utf8"))
                break
            client_socket.send("#".encode("utf8"))
            file_butter += data
        try:
            with open(options.upload, "wb") as f:
                f.write(file_butter.encode("utf8"))
            client_socket.send("save file success\n".encode("utf8"))
        except:
            client_socket.send("save file falied\n".encode("utf8"))
        finally:
            client_socket.close()

    #直接执行系统命令
    if options.execute:
        output = run_command(options.execute)
        client_socket.send(output)

    #返回一个shell 有一点输出bug
    if options.command:
        try:
            while True:
                client_socket.send("<BHP:#> ".encode("utf8"))
                cmd_buffer = ""
                while "\n" not in cmd_buffer:
                    try:
                        cmd_buffer += client_socket.recv(1024).decode("utf8")
                    except:
                        client_socket.close()
                        break
                response = run_command(cmd_buffer)
                client_socket.send(response)
        except:
            print("Failed")
        finally:
            client_socket.close()


#服务器循环，一直监听客户端接入
def server_loop():
    global options
    if not options.target:
        options.target = "0.0.0.0"
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.bind((options.target, options.port))
    print("Listen %s:%s" % (options.target, str(options.port)))
    server.listen(5)

    while True:
        client_socket, addr = server.accept()
        client_therad = threading.Thread(target=client_handler, args=(client_socket,))
        client_therad.start()


#客户端发送数据
def client_sender(buffer):
    client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    try:
        client.connect((options.target,options.port))
        if len(buffer):
            client.send(buffer.encode("utf8"))
        while True:
            recv_len = 1
            response = ""
            while recv_len:
                recv_len = client.recv(4096)
                try:
                    response += recv_len.decode("utf8")
                except:
                    response += recv_len.decode("gbk")
                if len(recv_len) < 4096:
                    break
            print(response + " ")   
            buffer = input("")
            buffer += "\n"
            client.send(buffer.encode("utf-8"))
    except:
        print("Failed to connect target")
    finally:
        client.close()


#主函数，判断监听还是连接
def main():
    global options
    options = init()
    if not options.listen and len(options.target) and options.port>0:
        buffer = input() + "\n"
        client_sender(buffer)

    if options.listen:
        server_loop()


if __name__ == "__main__":
    main()
